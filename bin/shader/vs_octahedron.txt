#version 440

in vec3 position;
in vec3 offset;
in float scale;

out PerVertex {
    vec3 position;
} outVertex;

uniform mat4 osg_ModelViewProjectionMatrix;

void main()
{
    outVertex.position = scale*position + offset;
    gl_Position = osg_ModelViewProjectionMatrix * vec4(outVertex.position, 1.0f);
}
